Ctrl+Alt+T     # Запуск терминала
dpkg --list    # Получить список всех установленных пакетов в системе
dpkg --list | grep "часть_названия" # Для поиска определенного пакета в данном списке, например, если ты не помнишь полное название, а лишь его часть можно воспользоватся командой grep и составить следующую конструкцию:
sudo apt-get remove vlc # Удаление приложения с сохранением персональных настроек (vlc программа)
sudo apt-get --purge remove vlc # Полная деинсталляция
sudo apt-get autoremove # Удалить зависимости
apt list --installed # Чтобы вывести список всех установленных пакетов используется команда:
snap list # Чтобы вывести список установленных Snap пакетов выполните команду:
sudo snap remove имяпакета # Для удаления Snap пакета выполните команду:
snap refresh # Обновление всех установленных пакетов:
snap refresh gimp # Обновление одного пакета:
snap find gimp # Для поиска пакетов используется команда snap find запрос
snap find "photo editor" # Поиск выполняется не только по имени пакета, но и по описанию, поэтому можно использовать произвольный запрос:
snap info gimp # Для получения информации о каком-либо пакете используется команда
snap revert gimp # Если по каким-то причинам вы хотите отменить обновления, которые были выполнены над каким-либо Snap-пакетом, то можно использовать команду snap revert имя_пакета, чтобы вернуть пакет к предыдущей версии.
snap disable gimp # Чтобы временно отключить пакет используется команда
snap enable gimp # Чтобы снова сделать пакет доступным используется команда 
snap run gimp # Для запуска Snap-пакета можно использовать команду
